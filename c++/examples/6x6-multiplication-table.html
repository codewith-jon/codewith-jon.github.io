C++ program to generate a 6x6 multiplication table

Example to print a n-by-n multiplication table arranged in horizontal rows and vertical columns (entered by the user) in C++ Programming. This problem is solved using nested for loop .


    /*------simple c++ program print n-by-n multiplication table----*/ 
#include <iostream>

using namespace std; 

int main()
{
    int b, j, x, y;

    cout<<"Enter the first operator \t";
    cin>>x;
    cout<<"Enter the second operator \t";
    cin>>y;
    for (b= 1; b <= x; b++)
     {
       for (j= 1; j <= y; j++)
        {
          cout<< b*j << "  ";
        }
      cout<<"\n";
     }
   return 0;
}
